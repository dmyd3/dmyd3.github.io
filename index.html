Exercício 4.1 - Equalização

Neste exercício uma imagem em escala de cinza será equalizada e então será destacado a diferença entre a imagem original e nova.

Primeiramente convertemos a imagem colorida para grayscale e então a equalizamos através da função cv::equalizeHist da seguinte forma:

Listagem 9. Codigo 4.1 -Equalização
//Converte a imagem para cinza e equaliza
        cvtColor(image, grayimage, CV_BGR2GRAY);
        equalizeHist(grayimage, eqimage);

Depois criamos os histogramas das imagens seguindo os passos do código base histograms.cpp e geramos o seguinte resultado:

Figure 4. Imagem original á esquerda e equalizada á direita

Notamos que o histograma da imagem original(de cor amarela) tem uma quantidade de pixels concentrada por volta do valor 128, e que o histograma da imagem equalizada (de cor vermelha) tem seus valores melhores distribuídos. E olhando para as imagens percebe-se que a imagem equalizada realçou mais os detalhes da figura.

Segue o código completo:

Listagem 10. Codigo 4.1 -Equalização de Histogramas
#include <iostream>
#include <opencv2/opencv.hpp>

using namespace cv;
using namespace std;

int main(int argc, char** argv) {
        Mat image, eqimage, grayimage;
        int width, height;
        VideoCapture cap;
        vector<Mat> planes, planesP, planesEq;
        Mat histP, histEq;
        int nbins = 64;
        float range[] = { 0, 256 };
        const float *histrange = { range };
        bool uniform = true;
        bool acummulate = false;

        cap.open(0);

        if (!cap.isOpened()) {
                cout << "cameras indisponiveis";
                return -1;
        }

        width = cap.get(CV_CAP_PROP_FRAME_WIDTH);
        height = cap.get(CV_CAP_PROP_FRAME_HEIGHT);

        cout << "largura = " << width << endl;
        cout << "altura  = " << height << endl;

        int histw = nbins, histh = nbins / 2;
        //Hists Preto e Equalizado
        Mat histImgP(histh, histw, CV_8UC3, Scalar(0));
        Mat histImgEq(histh, histw, CV_8UC3, Scalar(0));

        while (1) {
                cap >> image;

                //Converte a imagem para cinza e equaliza
                cvtColor(image, grayimage, CV_BGR2GRAY);
                equalizeHist(grayimage, eqimage);

                //HISTS preto e Eq
                split(grayimage, planesP);
                split(eqimage, planesEq);
                calcHist(&planesP[0], 1, 0, Mat(), histP, 1,
                                &nbins, &histrange,
                                uniform, acummulate);
                calcHist(&planesEq[0], 1, 0, Mat(), histEq, 1,
                                &nbins, &histrange,
                                uniform, acummulate);

                //Normalizando
                normalize(histP, histP, 0, histImgP.rows, NORM_MINMAX, -1, Mat());
                histImgP.setTo(Scalar(0));
                normalize(histEq, histEq, 0, histImgEq.rows, NORM_MINMAX, -1, Mat());
                histImgEq.setTo(Scalar(0));

                for (int i = 0; i<nbins; i++) {
                        line(histImgP,
                                Point(i, histh),
                                Point(i, histh - cvRound(histP.at<float>(i))),
                                Scalar(0,255,255), 1, 8, 0);
                        line(histImgEq,
                                Point(i, histh),
                                Point(i, histh - cvRound(histEq.at<float>(i))),
                                Scalar(0,0,255), 1, 8, 0);
                }

                //Ler as imagens de escala de cinza em RGB
                //para poder colocar seus histogramas
                cvtColor(eqimage, eqimage, CV_GRAY2RGB);
                cvtColor(grayimage, grayimage, CV_GRAY2RGB);

                //Coloca os histogramas no canto das imagens
                histImgP.copyTo(grayimage(Rect(0, 0, nbins, histh)));
                histImgEq.copyTo(eqimage(Rect(0, 0, nbins, histh)));

                imshow("image", image);
                imshow("Escala de Cinza", grayimage);
                imshow("Equalizado", eqimage);

                if (waitKey(30) >= 0) break;
        }
        return 0;
}

Segue abaixo o codigo completo:

Listagem 12. Codigo 4.1 -Equalização de Histogramas
#include <iostream>
#include <opencv2/opencv.hpp>
#include <math.h>

using namespace cv;
using namespace std;

int main(int argc, char** argv) {
        Mat image, eqimage, grayimage;
        int width, height, contadorM=0;
        VideoCapture cap;
        vector<Mat> planes;
        Mat histR, histG, histB, histEq, ant_histR;
        int nbins = 64;
        float range[] = { 0, 256 };
        const float *histrange = { range };
        bool uniform = true;
        bool acummulate = false;
        double Dcomp, ant_comp=0;

        cap.open(0);

        if (!cap.isOpened()) {
                cout << "cameras indisponiveis";
                return -1;
        }

        width = cap.get(CV_CAP_PROP_FRAME_WIDTH);
        height = cap.get(CV_CAP_PROP_FRAME_HEIGHT);

        cout << "largura = " << width << endl;
        cout << "altura  = " << height << endl;

        int histw = nbins, histh = nbins / 2;
        Mat histImgR(histh, histw, CV_8UC3, Scalar(0, 0, 0));
        Mat histImgG(histh, histw, CV_8UC3, Scalar(0, 0, 0));
        Mat histImgB(histh, histw, CV_8UC3, Scalar(0, 0, 0));

        while (1) {
                cap >> image;
                split(image, planes);
                calcHist(&planes[0], 1, 0, Mat(), histR, 1,
                        &nbins, &histrange,
                        uniform, acummulate);
                calcHist(&planes[1], 1, 0, Mat(), histG, 1,
                        &nbins, &histrange,
                        uniform, acummulate);
                calcHist(&planes[2], 1, 0, Mat(), histB, 1,
                        &nbins, &histrange,
                        uniform, acummulate);

                normalize(histR, histR, 0, histImgR.rows, NORM_MINMAX, -1, Mat());
                normalize(histG, histG, 0, histImgG.rows, NORM_MINMAX, -1, Mat());
                normalize(histB, histB, 0, histImgB.rows, NORM_MINMAX, -1, Mat());

                histImgR.setTo(Scalar(0));
                histImgG.setTo(Scalar(0));
                histImgB.setTo(Scalar(0));

                for (int i = 0; i<nbins; i++) {
                        line(histImgR,
                                Point(i, histh),
                                Point(i, histh - cvRound(histR.at<float>(i))),
                                Scalar(0, 0, 255), 1, 8, 0);
                        line(histImgG,
                                Point(i, histh),
                                Point(i, histh - cvRound(histG.at<float>(i))),
                                Scalar(0, 255, 0), 1, 8, 0);
                        line(histImgB,
                                Point(i, histh),
                                Point(i, histh - cvRound(histB.at<float>(i))),
                                Scalar(255, 0, 0), 1, 8, 0);
                }

                if(ant_histR.empty()) ant_histR = histR;

                //Dcomp = compareHist(histR, ant_histR, CV_COMP_CORREL);
                Dcomp = compareHist(histR, ant_histR, CV_COMP_INTERSECT);

                ant_histR = histR; //O atual vira o anterior pós comparaçao

                if (fabs(Dcomp - ant_comp) > 55) {
                        contadorM++;
                        cout << "Movimento: " << contadorM << endl;;
                        cout << "Diferença: " << fabs(Dcomp - ant_comp) << endl;
                        //Desenha circulo amarelo na imagem
                        circle(image, Point(128, 64), 50, Scalar(0, 255, 255), -1, 8, 0);
                }
                ant_comp = Dcomp; //Comparaçao atual vira a anterior


                histImgR.copyTo(image(Rect(0, 0, nbins, histh)));
                histImgG.copyTo(image(Rect(0, histh, nbins, histh)));
                histImgB.copyTo(image(Rect(0, 2 * histh, nbins, histh)));
                imshow("image", image);

                if (waitKey(30) >= 0) break;
        }
        return 0;
}

Essa funçao chama outra função que tambem é chamada pelos trackbars de posição e altura da região focada da imagem.

Essa função define a regiao da imagem que estará em foque. Após a execução desses 2 métodos é hora de juntar a regiao focada e borrada em apenas uma imagem, usando cv::addWeighted, mas para isso precisamos definir o alfa pela seguinte equação:

No código o alfa é calculado da seguinte forma:

void on_trackbar_blend(int, void*) {

        blended = image1.clone();
        n1 = inicio;
        n2 = inicio + limit;
        d = (desfoque / 20)*100; //fazendo a variavel desfoque entre 0 a 100

        for (int i = 0; i<blended.rows; i++) {
                alfa= 0.5*(tanh((i-n1)/(d+ 1)) - tanh((i-n2)/(d+ 1)));
                beta = 1 - alfa;
                addWeighted(imageTop.row(i), alfa, imageBlur.row(i), beta, 0.0, blended.row(i));
        }
        imshow("addweighted", blended);
}
